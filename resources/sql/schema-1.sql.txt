Zur Dokumentation! Tabellen-Prefix "wt_" ist ggf. auch anders definiert!

CREATE TABLE wt_transcription_metadata (
                                           setting_name  VARCHAR(64) PRIMARY KEY,
                                           setting_value TEXT NOT NULL
);

CREATE TABLE wt_transcriptions (
                                   id                  INT AUTO_INCREMENT PRIMARY KEY,
                                   tree_id             INT NOT NULL,
                                   source_xref         VARCHAR(20) NOT NULL,
                                   media_xref          VARCHAR(20) NULL,
                                   title               VARCHAR(255) NOT NULL,
                                   transcription_type  VARCHAR(32) NOT NULL,
                                   provider_key        VARCHAR(32) NOT NULL,
                                   status              VARCHAR(32) NOT NULL,
                                   tag_note_xref       VARCHAR(20) NULL,
                                   current_note_xref   VARCHAR(20) NULL,
                                   created_by_user_id  INT NOT NULL,
                                   created_at          TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
                                   updated_at          TIMESTAMP NULL DEFAULT NULL,
                                   is_active           TINYINT(1) NOT NULL DEFAULT 1
);

CREATE TABLE wt_transcription_revisions (
                                            id                    INT AUTO_INCREMENT PRIMARY KEY,
                                            transcription_id      INT NOT NULL,
                                            revision_no           INT NOT NULL,
                                            provider_key          VARCHAR(32) NOT NULL,
                                            origin_type           VARCHAR(32) NOT NULL,
                                            origin_reference      TEXT NULL,
                                            content_format        VARCHAR(32) NOT NULL,
                                            content_text          LONGTEXT NOT NULL,
                                            content_hash          CHAR(64) NOT NULL,
                                            created_by_user_id    INT NOT NULL,
                                            created_at            TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
                                            import_comment        TEXT NULL,
                                            generated_note_xref   VARCHAR(20) NULL,
                                            is_current_revision   TINYINT(1) NOT NULL DEFAULT 0
);

CREATE TABLE wt_transcription_note_links (
                                             id                      INT AUTO_INCREMENT PRIMARY KEY,
                                             transcription_id        INT NOT NULL,
                                             revision_id             INT NULL,
                                             note_xref               VARCHAR(20) NOT NULL,
                                             link_type               VARCHAR(32) NOT NULL,
                                             created_at              TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
                                             created_by_user_id      INT NOT NULL,
                                             is_current              TINYINT(1) NOT NULL DEFAULT 0,
                                             note_hash_at_link_time  CHAR(64) NULL
);
